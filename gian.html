using System; using System.Collections.Generic; using System.IO; using System.Globalization; class Homework { public string Subject { get; set; } public string Description { get; set; } public DateTime Deadline { get; set; } public bool IsCompleted { get; set; } } class Program { static List tasks = new List(); static readonly string filePath = "homework.txt"; static void Main() { LoadFromFile(); CheckDeadlines(); ShowMenu(); SaveToFile(); } // ================= MENU ================= static void ShowMenu() { string choice; do { Console.Clear(); Console.WriteLine("==== 📚 HOMEWORK TRACKER PRO ===="); Console.WriteLine("1. Add Homework"); Console.WriteLine("2. View Homework"); Console.WriteLine("3. Delete Homework"); Console.WriteLine("4. Mark as Completed"); Console.WriteLine("5. Search Homework"); Console.WriteLine("0. Exit"); Console.Write("Enter choice: "); choice = Console.ReadLine(); switch (choice) { case "1": AddHomework(); break; case "2": ViewHomework(); break; case "3": DeleteHomework(); break; case "4": MarkCompleted(); break; case "5": SearchHomework(); break; case "0": Console.WriteLine("Saving and exiting..."); break; default: Console.WriteLine("Invalid choice."); break; } Pause(); } while (choice != "0"); } // ================= ADD ================= static void AddHomework() { Homework hw = new Homework(); hw.Subject = ReadInput("Enter Subject: "); hw.Description = ReadInput("Enter Description: "); hw.Deadline = ReadDate("Enter Deadline (YYYY-MM-DD): "); hw.IsCompleted = false; tasks.Add(hw); Console.WriteLine("✅ Homework added!"); } // ================= VIEW ================= static void ViewHomework() { Console.WriteLine("\n==== 📋 HOMEWORK LIST ===="); if (tasks.Count == 0) { Console.WriteLine("No tasks available."); return; } for (int i = 0; i < tasks.Count; i++) { var hw = tasks[i]; SetColor(hw); string status = hw.IsCompleted ? "[✔]" : "[ ]"; Console.WriteLine($"{i + 1}. {status} {hw.Subject} - {hw.Description} (Due: {hw.Deadline:yyyy-MM-dd})"); Console.ResetColor(); } } // ================= DELETE ================= static void DeleteHomework() { ViewHomework(); if (tasks.Count == 0) return; int index = ReadIndex("Enter ID to delete: "); Console.Write("Are you sure? (y/n): "); if (Console.ReadLine().ToLower() == "y") { tasks.RemoveAt(index); Console.WriteLine("🗑️ Deleted successfully."); } } // ================= COMPLETE ================= static void MarkCompleted() { ViewHomework(); if (tasks.Count == 0) return; int index = ReadIndex("Enter ID to mark as completed: "); tasks[index].IsCompleted = true; Console.WriteLine("🎯 Marked as completed!"); } // ================= SEARCH ================= static void SearchHomework() { string key = ReadInput("Search keyword: ").ToLower(); var results = tasks.FindAll(h => h.Subject.ToLower().Contains(key) || h.Description.ToLower().Contains(key)); Console.WriteLine("\n==== 🔍 RESULTS ===="); if (results.Count == 0) { Console.WriteLine("No results found."); return; } foreach (var hw in results) { Console.WriteLine($"{(hw.IsCompleted ? "[✔]" : "[ ]")} {hw.Subject} (Due: {hw.Deadline:yyyy-MM-dd})"); } } // ================= DEADLINE CHECK ================= static void CheckDeadlines() { Console.WriteLine("\n🔔 DEADLINE CHECK"); DateTime today = DateTime.Today; bool hasAlert = false; foreach (var hw in tasks) { if (hw.IsCompleted) continue; double days = (hw.Deadline - today).TotalDays; if (days < 0) { Console.ForegroundColor = ConsoleColor.Red; Console.WriteLine($"OVERDUE: {hw.Subject}"); hasAlert = true; } else if (days <= 1) { Console.ForegroundColor = ConsoleColor.Yellow; Console.WriteLine($"URGENT: {hw.Subject} (Due soon!)"); hasAlert = true; } } if (!hasAlert) Console.WriteLine("✅ All tasks are on track!"); Console.ResetColor(); Pause(); } // ================= FILE HANDLING ================= static void SaveToFile() { try { using StreamWriter sw = new StreamWriter(filePath); foreach (var hw in tasks) { sw.WriteLine($"{hw.Subject}|{hw.Description}|{hw.Deadline:yyyy-MM-dd}|{hw.IsCompleted}"); } } catch (Exception ex) { Console.WriteLine("Error saving file: " + ex.Message); } } static void LoadFromFile() { if (!File.Exists(filePath)) return; try { foreach (var line in File.ReadAllLines(filePath)) { string[] d = line.Split('|'); if (d.Length < 4) continue; tasks.Add(new Homework { Subject = d[0], Description = d[1], Deadline = DateTime.ParseExact(d[2], "yyyy-MM-dd", CultureInfo.InvariantCulture), IsCompleted = bool.Parse(d[3]) }); } } catch { Console.WriteLine("⚠️ Error loading file."); } } // ================= HELPERS ================= static string ReadInput(string message) { string input; do { Console.Write(message); input = Console.ReadLine(); } while (string.IsNullOrWhiteSpace(input)); return input; } static DateTime ReadDate(string message) { DateTime date; while (true) { Console.Write(message); if (DateTime.TryParseExact(Console.ReadLine(), "yyyy-MM-dd", CultureInfo.InvariantCulture, DateTimeStyles.None, out date)) return date; Console.WriteLine("❌ Invalid format. Use YYYY-MM-DD."); } } static int ReadIndex(string message) { int index; while (true) { Console.Write(message); if (int.TryParse(Console.ReadLine(), out index) && index > 0 && index <= tasks.Count) return index - 1; Console.WriteLine("Invalid selection."); } } static void SetColor(Homework hw) { if (hw.IsCompleted) Console.ForegroundColor = ConsoleColor.Green; else if (hw.Deadline < DateTime.Today) Console.ForegroundColor = ConsoleColor.Red; } static void Pause() { Console.WriteLine("\nPress any key to continue..."); Console.ReadKey(); } }
